# JS技巧

一些js基础技巧

# 生成1-20的数组，不用for循环

  • keys+slice
[...Array(21).keys()].slice(1)
  • Array.from
Array.from({length: 20}, (v, k) => k + 1)

# 数据去重

  • Set去重
  const arr = [1, 9, 3, 9, 1, 1, 4, 3]
  [...new Set(arr)]   // [1, 9, 3, 4]
  • filter+indexOf
  const arr = [1, 9, 3, 9, 1, 1, 4, 3]
  arr.filter((e, i, _arr) => _arr.indexOf(e) === i)   // [1, 9, 3, 4]
  • reduce+includes
  const arr = [1, 9, 3, 9, 1, 1, 4, 3]
  arr.reduce((acc, cur) => acc.includes(cur) ? acc : [...acc, cur], [])  // [1, 9, 3 ,4]
